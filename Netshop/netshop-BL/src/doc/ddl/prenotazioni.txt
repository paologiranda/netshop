CREATE TABLE  "PRENOTAZIONI" 
   (	"IDPRENOTAZIONE" NUMBER(5,0), 
	"CODPRODOTTO" VARCHAR2(12) NOT NULL ENABLE, 
	"CODCLIENTE" NUMBER(5,0) NOT NULL ENABLE, 
	"DATAPRENOTAZIONE" DATE NOT NULL ENABLE, 
	"DATAARRIVO" DATE NOT NULL ENABLE, 
	"QTA" NUMBER(5,0) NOT NULL ENABLE, 
	 CONSTRAINT "PRENOTAZIONI_PK" PRIMARY KEY ("IDPRENOTAZIONE") ENABLE
   ) ;

CREATE OR REPLACE TRIGGER  BI_PRENOTAZIONI 
  before insert on PRENOTAZIONI               
  for each row  
begin   
  if :NEW.IDPRENOTAZIONE is null then 
    select PRENOTAZIONI_SEQ.nextval into :NEW.IDPRENOTAZIONE from dual; 
  end if; 
end;
